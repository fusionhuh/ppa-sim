	reg [`WIDTH-1:0]x2;
	wire [`WIDTH-1:0]s;

	wire cout;
	{adder} test(.x1(x1), .x2(x2), .s(s), .cin(cin), .cout(cout));

	initial begin

{tests}
		$display("All tests passed!");
		$finish;
	end
